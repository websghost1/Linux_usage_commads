#list all usb devices:
lsusb

#Watch command

watch -d lsusb

ברגע שמריצים את הפקודה הזאת
 המערכת תבדוק האם בוצע חיבור חדש של התקן יו אס בי למערכת


#להציג איזה יו אס בי מחובר לאיזה פורט
lsbusb -t


#לבדוק את פרטי המחיצה של דיסק ה-USB
sudo blkid /dev/sdX1
הפקודה מציגה את סוג מערכת הקבצים של המחיצה ואת ה-UUID שלה.

#בלינוקס צריך להגיד למערכת לאן למפות את הדיסק און קי בשביל שאפשר יהיה לגשת אליו
1. אז קודם צריך ליצור תיקייה בכל מקום - נהגור בנתיב הזה
sudo mkdir /mnt/usb

2. לאחר מכן נחבר את הדיסק און קי לתיקייה שיצרנו
sudo mount /dev/sdX1 /mnt/usb
לאחר ההרצה, התוכן של דיסק ה-USB יוצג בתוך /mnt/usb.


#לנתק את דיסק ה-USB בצורה תקינה
sudo umount /mnt/usb
רק לאחר מכן ניתן לנתק פיזית את הדיסק מהמחשב.
